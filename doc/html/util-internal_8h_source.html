00193 /* Replacement for assert() that calls event_errx on failure. */
00194 #ifdef NDEBUG
00195 #define EVUTIL_ASSERT(cond) _EVUTIL_NIL_CONDITION(cond)
00196 #define EVUTIL_FAILURE_CHECK(cond) 0
00197 #else
00198 #define EVUTIL_ASSERT(cond)                                             \
00199         do {                                                            \
00200                 if (EVUTIL_UNLIKELY(!(cond))) {                         \
00201                         event_errx(_EVENT_ERR_ABORT,                    \
00202                             "%s:%d: Assertion %s failed in %s",         \
00203                             __FILE__,__LINE__,#cond,__func__);          \
00204                         /* In case a user-supplied handler tries to */  \
00205                         /* return control to us, log and abort here. */ \
00206                         (void)fprintf(stderr,                           \
00207                             "%s:%d: Assertion %s failed in %s",         \
00208                             __FILE__,__LINE__,#cond,__func__);          \
00209                         abort();                                        \
00210                 }                                                       \
00211         } while (0)
00212 #define EVUTIL_FAILURE_CHECK(cond) EVUTIL_UNLIKELY(cond)
00213 #endif
