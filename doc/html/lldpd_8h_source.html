00276 /* LLDP: "ether proto 0x88cc and ether dst 01:80:c2:00:00:0e" */
00277 /* FDP: "ether dst 01:e0:52:cc:cc:cc" */

00279 /* SONMP: "ether dst 01:00:81:00:01:00" */
00280 /* EDP: "ether dst 00:e0:2b:00:00:00" */
00281 /* For optimization purpose, we first check if the first bit of the
00282    first byte is 1. if not, this can only be an EDP packet:
00283 
00284    tcpdump -dd "(ether[0] & 1 = 1 and

00286                   (ether dst 01:e0:52:cc:cc:cc) or
00287                   (ether dst 01:00:0c:cc:cc:cc) or
00288                   (ether dst 01:00:81:00:01:00))) or
00289                 (ether dst 00:e0:2b:00:00:00)"
00290 */
