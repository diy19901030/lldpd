00029 /*
00030   A "changelist" is a list of all the fd status changes that should be made
00031   between calls to the backend's dispatch function.  There are a few reasons
00032   that a backend would want to queue changes like this rather than processing
00033   them immediately.
00034 
00035     1) Sometimes applications will add and delete the same event more than
00036        once between calls to dispatch.  Processing these changes immediately
00037        is needless, and potentially expensive (especially if we're on a system
00038        that makes one syscall per changed event).
00039 
00040     2) Sometimes we can coalesce multiple changes on the same fd into a single
00041        syscall if we know about them in advance.  For example, epoll can do an
00042        add and a delete at the same time, but only if we have found out about
00043        both of them before we tell epoll.
00044 
00045     3) Sometimes adding an event that we immediately delete can cause
00046        unintended consequences: in kqueue, this makes pending events get
00047        reported spuriously.
00048  */

00053 struct event_change {
00055         evutil_socket_t fd;
00056         /* The events that were enabled on the fd before any of these changes
00057            were made.  May include EV_READ or EV_WRITE. */
00058         short old_events;
00059 
00060         /* The changes that we want to make in reading and writing on this fd.
00061          * If this is a signal, then read_change has EV_CHANGE_SIGNAL set,
00062          * and write_change is unused. */
00063         ev_uint8_t read_change;
00064         ev_uint8_t write_change;
00065 };

00067 /* Flags for read_change and write_change. */
00068 
00069 /* If set, add the event. */
00070 #define EV_CHANGE_ADD     0x01
00071 /* If set, delete the event.  Exclusive with EV_CHANGE_ADD */
00072 #define EV_CHANGE_DEL     0x02
00073 /* If set, this event refers a signal, not an fd. */
00074 #define EV_CHANGE_SIGNAL  EV_SIGNAL
00075 /* Set for persistent events.  Currently not used. */
00076 #define EV_CHANGE_PERSIST EV_PERSIST
00077 /* Set for adding edge-triggered events. */
00078 #define EV_CHANGE_ET      EV_ET
