00027 /* This set of macro are used to build packets. The current position in buffer
00028  * is `pos'. The length of the remaining space in buffer is `length'. `type'
00029  * should be a member of `types'. This was stolen from ladvd which was adapted
00030  * from Net::CDP. */
00031 
00032 #define POKE(value, type, func)                   \
00033         ((length >= sizeof(type)) &&              \
00034             (                                     \
00035                 type = func(value),               \
00036                 memcpy(pos, &type, sizeof(type)), \
00037                 length -= sizeof(type),           \
00038                 pos += sizeof(type),              \
00039                 1                                 \
00040             ))
00041 #define POKE_UINT8(value) POKE(value, types.f_uint8, )
00042 #define POKE_UINT16(value) POKE(value, types.f_uint16, htons)
00043 #define POKE_UINT32(value) POKE(value, types.f_uint32, htonl)
00044 #define POKE_BYTES(value, bytes)                               \
00045         ((length >= (bytes)) &&                                \
00046             (                                                  \
00047                 memcpy(pos, value, bytes),                     \
00048                 length -= (bytes),                             \
00049                 pos += (bytes),                                \
00050                 1                                              \
00051             ))
00052 #define POKE_SAVE(where)                        \
00053         (where = pos, 1)
00054 #define POKE_RESTORE(where)                     \
00055         do {                                    \
00056         if ((where) > pos)                      \
00057                 length -= ((where) - pos);      \
00058         else                                    \
00059                 length += (pos - (where));      \
00060         pos = (where);                          \
00061         } while(0)

00063 /* This set of macro are used to parse packets. The same variable as for POKE_*
00064  * are used. There is no check on boundaries. */

00066 #define PEEK(type, func)                                \
00067         (                                               \
00068                 memcpy(&type, pos, sizeof(type)),       \
00069                 length -= sizeof(type),                 \
00070                 pos += sizeof(type),                    \
00071                 func(type)                              \
00072         )
00073 #define PEEK_UINT8 PEEK(types.f_uint8, )
00074 #define PEEK_UINT16 PEEK(types.f_uint16, ntohs)
00075 #define PEEK_UINT32 PEEK(types.f_uint32, ntohl)
00076 #define PEEK_BYTES(value, bytes)                               \
00077         do {                                                   \
00078                 memcpy(value, pos, bytes);                     \
00079                 length -= (bytes);                             \
00080                 pos += (bytes);                                \
00081         } while (0)
00082 #define PEEK_DISCARD(bytes)                     \
00083         do {                                    \
00084                 length -= (bytes);              \
00085                 pos += (bytes);                 \
00086         } while (0)
00087 #define PEEK_DISCARD_UINT8 PEEK_DISCARD(1)
00088 #define PEEK_DISCARD_UINT16 PEEK_DISCARD(2)
00089 #define PEEK_DISCARD_UINT32 PEEK_DISCARD(3)
00090 #define PEEK_CMP(value, bytes)                  \
00091         (length -= (bytes),                     \
00092          pos += (bytes),                        \
00093          memcmp(pos-bytes, value, bytes))
00094 #define PEEK_SAVE POKE_SAVE
00095 #define PEEK_RESTORE POKE_RESTORE
