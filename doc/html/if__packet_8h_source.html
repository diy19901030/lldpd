00157 struct tpacket_hdr_v1 {
00158         __u32   block_status;
00159         __u32   num_pkts;
00160         __u32   offset_to_first_pkt;
00161 
00162         /* Number of valid bytes (including padding)
00163          * blk_len <= tp_block_size
00164          */
00165         __u32   blk_len;
00166 
00167         /*
00168          * Quite a few uses of sequence number:
00169          * 1. Make sure cache flush etc worked.
00170          *    Well, one can argue - why not use the increasing ts below?
00171          *    But look at 2. below first.
00172          * 2. When you pass around blocks to other user space decoders,
00173          *    you can see which blk[s] is[are] outstanding etc.
00174          * 3. Validate kernel code.
00175          */
00176         __aligned_u64   seq_num;
00177 
00178         /*
00179          * ts_last_pkt:
00180          *
00181          * Case 1.      Block has 'N'(N >=1) packets and TMO'd(timed out)
00182          *              ts_last_pkt == 'time-stamp of last packet' and NOT the
00183          *              time when the timer fired and the block was closed.
00184          *              By providing the ts of the last packet we can absolutely
00185          *              guarantee that time-stamp wise, the first packet in the
00186          *              next block will never precede the last packet of the
00187          *              previous block.
00188          * Case 2.      Block has zero packets and TMO'd
00189          *              ts_last_pkt = time when the timer fired and the block
00190          *              was closed.
00191          * Case 3.      Block has 'N' packets and NO TMO.
00192          *              ts_last_pkt = time-stamp of the last pkt in the block.
00193          *
00194          * ts_first_pkt:
00195          *              Is always the time-stamp when the block was opened.
00196          *              Case a) ZERO packets
00197          *                      No packets to deal with but atleast you know the
00198          *                      time-interval of this block.
00199          *              Case b) Non-zero packets
00200          *                      Use the ts of the first packet in the block.
00201          *
00202          */
00203         struct tpacket_bd_ts    ts_first_pkt, ts_last_pkt;
00204 };
