00020 struct ethtool_cmd {
00021         __u32   cmd;
00022         __u32   supported;      /* Features this interface supports */
00023         __u32   advertising;    /* Features this interface advertises */
00024         __u16   speed;          /* The forced speed (lower bits) in
00025                                  * Mbps. Please use
00026                                  * ethtool_cmd_speed()/_set() to
00027                                  * access it */
00028         __u8    duplex;         /* Duplex, half or full */
00029         __u8    port;           /* Which connector port */
00030         __u8    phy_address;    /* MDIO PHY address (PRTAD for clause 45).
00031                                  * May be read-only or read-write
00032                                  * depending on the driver.
00033                                  */
00034         __u8    transceiver;    /* Which transceiver to use */
00035         __u8    autoneg;        /* Enable or disable autonegotiation */
00036         __u8    mdio_support;   /* MDIO protocols supported.  Read-only.
00037                                  * Not set by all drivers.
00038                                  */
00039         __u32   maxtxpkt;       /* Tx pkts before generating tx int */
00040         __u32   maxrxpkt;       /* Rx pkts before generating rx int */
00041         __u16   speed_hi;       /* The forced speed (upper
00042                                  * bits) in Mbps. Please use
00043                                  * ethtool_cmd_speed()/_set() to
00044                                  * access it */
00045         __u8    eth_tp_mdix;
00046         __u8    reserved2;
00047         __u32   lp_advertising; /* Features the link partner advertises */
00048         __u32   reserved[2];
00049 };

00051 static __inline__ void ethtool_cmd_speed_set(struct ethtool_cmd *ep,
00052                                          __u32 speed)
00053 {
00054 
00055         ep->speed = (__u16)speed;
00056         ep->speed_hi = (__u16)(speed >> 16);
00057 }
00058 
00059 static __inline__ __u32 ethtool_cmd_speed(const struct ethtool_cmd *ep)
00060 {
00061         return (ep->speed_hi << 16) | ep->speed;
00062 }

00776 /*
00777  * %ETHTOOL_SFEATURES changes features present in features[].valid to the
00778  * values of corresponding bits in features[].requested. Bits in .requested
00779  * not set in .valid or not changeable are ignored.
00780  *
00781  * Returns %EINVAL when .valid contains undefined or never-changeable bits
00782  * or size is not equal to required number of features words (32-bit blocks).
00783  * Returns >= 0 if request was completed; bits set in the value mean:
00784  *   %ETHTOOL_F_UNSUPPORTED - there were bits set in .valid that are not
00785  *      changeable (not present in %ETHTOOL_GFEATURES' features[].available)
00786  *      those bits were ignored.
00787  *   %ETHTOOL_F_WISH - some or all changes requested were recorded but the
00788  *      resulting state of bits masked by .valid is not equal to .requested.
00789  *      Probably there are other device-specific constraints on some features
00790  *      in the set. When %ETHTOOL_F_UNSUPPORTED is set, .valid is considered
00791  *      here as though ignored bits were cleared.
00792  *   %ETHTOOL_F_COMPAT - some or all changes requested were made by calling
00793  *      compatibility functions. Requested offload state cannot be properly
00794  *      managed by kernel.
00795  *
00796  * Meaning of bits in the masks are obtained by %ETHTOOL_GSSET_INFO (number of
00797  * bits in the arrays - always multiple of 32) and %ETHTOOL_GSTRINGS commands
00798  * for ETH_SS_FEATURES string set. First entry in the table corresponds to least
00799  * significant bit in features[0] fields. Empty strings mark undefined features.
00800  */
00801 enum ethtool_sfeatures_retval_bits {
00802         ETHTOOL_F_UNSUPPORTED__BIT,
00803         ETHTOOL_F_WISH__BIT,
00804         ETHTOOL_F_COMPAT__BIT,
00805 };
00806 
00807 #define ETHTOOL_F_UNSUPPORTED   (1 << ETHTOOL_F_UNSUPPORTED__BIT)
00808 #define ETHTOOL_F_WISH          (1 << ETHTOOL_F_WISH__BIT)
00809 #define ETHTOOL_F_COMPAT        (1 << ETHTOOL_F_COMPAT__BIT)
