00041 /* These flags are reasons that we might be declining to actually enable
00042    reading or writing on a bufferevent.

00045 /* On a all bufferevents, for reading: used when we have read up to the
00046    watermark value.
00047 
00048    On a filtering bufferevent, for writing: used when the underlying
00049    bufferevent's write buffer has been filled up to its watermark
00050    value.
00051 */
00052 #define BEV_SUSPEND_WM 0x01
00053 /* On a base bufferevent: when we have emptied a bandwidth buckets */
00054 #define BEV_SUSPEND_BW 0x02
00055 /* On a base bufferevent: when we have emptied the group's bandwidth bucket. */
00056 #define BEV_SUSPEND_BW_GROUP 0x04
00057 /* On a socket bufferevent: can't do any operations while we're waiting for
00058  * name lookup to finish. */
00059 #define BEV_SUSPEND_LOOKUP 0x08
00060 /* On a base bufferevent, for reading: used when a filter has choked this
00061  * (underlying) bufferevent because it has stopped reading from it. */
00062 #define BEV_SUSPEND_FILT_READ 0x10
