00071 int min_heap_push(min_heap_t* s, struct event* e)
00072 {
00073         if (min_heap_reserve(s, s->n + 1))
00074                 return -1;

00096 int min_heap_erase(min_heap_t* s, struct event* e)
00097 {
00098         if (-1 != e->ev_timeout_pos.min_heap_idx)
00099         {
00100                 struct event *last = s->p[--s->n];
00101                 unsigned parent = (e->ev_timeout_pos.min_heap_idx - 1) / 2;
00102                 /* we replace e with the last element in the heap.  We might need to
00103                    shift it upward if it is less than its parent, or downward if it is
00104                    greater than one or both its children. Since the children are known
00105                    to be less than the parent, it can't need to shift both up and
00106                    down. */
00107                 if (e->ev_timeout_pos.min_heap_idx > 0 && min_heap_elem_greater(s->p[parent], last))
00108                         min_heap_shift_up_(s, e->ev_timeout_pos.min_heap_idx, last);
00109                 else
00110                         min_heap_shift_down_(s, e->ev_timeout_pos.min_heap_idx, last);
00111                 e->ev_timeout_pos.min_heap_idx = -1;
00112                 return 0;
00113         }
00114         return -1;
00115 }

00117 int min_heap_reserve(min_heap_t* s, unsigned n)
00118 {
00119         if (s->a < n)
00120         {
00121                 struct event** p;
00122                 unsigned a = s->a ? s->a * 2 : 8;
00123                 if (a < n)
00124                         a = n;
00125                 if (!(p = (struct event**)mm_realloc(s->p, a * sizeof *p)))
00126                         return -1;
00127                 s->p = p;
00128                 s->a = a;
00129         }
00130         return 0;
00131 }

00133 void min_heap_shift_up_(min_heap_t* s, unsigned hole_index, struct event* e)
00134 {
00135     unsigned parent = (hole_index - 1) / 2;
00136     while (hole_index && min_heap_elem_greater(s->p[parent], e))
00137     {
00138         (s->p[hole_index] = s->p[parent])->ev_timeout_pos.min_heap_idx = hole_index;
00139         hole_index = parent;
00140         parent = (hole_index - 1) / 2;
00141     }
00142     (s->p[hole_index] = e)->ev_timeout_pos.min_heap_idx = hole_index;
00143 }
00144 
00145 void min_heap_shift_down_(min_heap_t* s, unsigned hole_index, struct event* e)
00146 {
00147     unsigned min_child = 2 * (hole_index + 1);
00148     while (min_child <= s->n)
00149         {
00150         min_child -= min_child == s->n || min_heap_elem_greater(s->p[min_child], s->p[min_child - 1]);
00151         if (!(min_heap_elem_greater(e, s->p[min_child])))
00152             break;
00153         (s->p[hole_index] = s->p[min_child])->ev_timeout_pos.min_heap_idx = hole_index;
00154         hole_index = min_child;
00155         min_child = 2 * (hole_index + 1);
00156         }
00157     (s->p[hole_index] = e)->ev_timeout_pos.min_heap_idx = hole_index;
00158 }
