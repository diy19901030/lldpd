00299 /* Smart mode / Hide mode */
00300 #define SMART_INCOMING_FILTER           (1<<0) /* Incoming filtering enabled */
00301 #define SMART_INCOMING_ONE_PROTO        (1<<1) /* On reception, keep only one proto */
00302 #define SMART_INCOMING_ONE_NEIGH        (1<<2) /* On reception, keep only one neighbor */
00303 #define SMART_OUTGOING_FILTER           (1<<3) /* Outgoing filtering enabled */
00304 #define SMART_OUTGOING_ONE_PROTO        (1<<4) /* On emission, keep only one proto */
00305 #define SMART_OUTGOING_ONE_NEIGH        (1<<5) /* On emission, consider only one neighbor */
00306 #define SMART_INCOMING (SMART_INCOMING_FILTER |    \
00307                          SMART_INCOMING_ONE_PROTO | \
00308                          SMART_INCOMING_ONE_NEIGH)
00309 #define SMART_OUTGOING (SMART_OUTGOING_FILTER |         \
