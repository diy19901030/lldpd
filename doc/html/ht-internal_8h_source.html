00058 static inline unsigned
00059 ht_improve_hash(unsigned h)
00060 {
00061   /* Aim to protect against poor hash functions by adding logic here
00062    * - logic taken from java 1.4 hashtable source */
00063   h += ~(h << 9);
00064   h ^=  ((h >> 14) | (h << 18)); /* >>> */
00065   h +=  (h << 4);
00066   h ^=  ((h >> 10) | (h << 22)); /* >>> */
00067   return h;
00068 }

00086 static inline unsigned
00087 ht_string_hash(const char *s)
00088 {
00089   unsigned h;
00090   const unsigned char *cp = (const unsigned char *)s;
00091   h = *cp << 7;
00092   while (*cp) {
00093     h = (1000003*h) ^ *cp++;
00094   }
00095   /* This conversion truncates the length of the string, but that's ok. */
00096   h ^= (unsigned)(cp-(const unsigned char*)s);
00097   return h;
00098 }

00324       new_len = name##_PRIMES[++prime_idx];                             \
